<script>

    let correoValidado = "";

    let montoValidado = 0;


    async function verificarUsuario() {

        const email = document.getElementById('email').value;

        const amount = document.getElementById('amount').value;

        const errorMsg = document.getElementById('error-message');

        const btnVerify = document.getElementById('btn-verify');


        if(!email || !amount) {

            mostrarError("Por favor completa todos los campos.");

            return;

        }


        btnVerify.innerText = "Verificando...";

        btnVerify.disabled = true;

        errorMsg.style.display = "none";


        try {

            // Llamada al Backend Real

            const response = await fetch('/api/check-user', {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify({ email: email })

            });

            

            const data = await response.json();


            if (data.existe) {

                // Éxito: El usuario existe en la DB

                correoValidado = email;

                montoValidado = amount;

                

                btnVerify.style.display = 'none'; // Ocultar botón verificar

                document.getElementById('email').disabled = true;

                document.getElementById('amount').disabled = true;

                

                document.getElementById('paypal-button-container').style.display = "block";

                renderizarPayPal();

            } else {

                mostrarError("Este correo no está registrado en nuestra base de datos.");

                resetBtn();

            }

        } catch (error) {

            console.error(error);

            mostrarError("Error de conexión con el servidor.");

            resetBtn();

        }

    }


    function mostrarError(texto) {

        const el = document.getElementById('error-message');

        el.innerText = texto;

        el.style.display = "block";

    }


    function resetBtn() {

        const btn = document.getElementById('btn-verify');

        btn.innerText = "Verificar Cuenta";

        btn.disabled = false;

    }


    function renderizarPayPal() {

        paypal.Buttons({

            createOrder: function(data, actions) {

                return actions.order.create({

                    purchase_units: [{

                        amount: { value: montoValidado },

                        description: `Créditos IPPBLOCK para: ${correoValidado}`

                    }]

                });

            },

            onApprove: function(data, actions) {

                return actions.order.capture().then(function(details) {

                    alert('Pago realizado con éxito por ' + details.payer.name.given_name);

                    

                    // Notificar al servidor para enviar correos

                    fetch('/api/payment-success', {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify({

                            email: correoValidado,

                            amount: montoValidado,

                            orderID: data.orderID,

                            payerName: details.payer.name.given_name

                        })

                    });

                });

            }

        }).render('#paypal-button-container');

    }

</script>
